set(CMAKE_CXX_STANDARD		    23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS		OFF)

file(GLOB CERO_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/**/*.hpp ./**/*.cpp)
add_executable(Cero ${CERO_SOURCES})

target_include_directories(Cero PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})

source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${CERO_SOURCES})

set_property(TARGET Cero PROPERTY PRECOMPILE_HEADERS "PCH.hpp")
file(GLOB_RECURSE PCH_FILES ${CMAKE_CURRENT_SOURCE_DIR}${CMAKE_FILES_DIRECTORY}/cmake_pch.*)
source_group("generated" FILES ${PCH_FILES})
